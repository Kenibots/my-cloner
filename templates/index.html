import discord from discord.ext import commands from flask import Flask, render_template, request, jsonify import threading import asyncio app = Flask(__name__) async def start_cloning(user_token, source_id, target_id): # discord.py-self кітапханасы немесе ескі нұсқалар self_bot-ты қолдайды # МАҢЫЗДЫ: Стандартты discord.py self-bot-қа шектеу қоюы мүмкін bot = commands.Bot(command_prefix=".", self_bot=True) @bot.event async def on_ready(): print(f'Аккаунтқа кірдік: {bot.user}') try: source_guild = bot.get_guild(int(source_id)) target_guild = bot.get_guild(int(target_id)) if source_guild and target_guild: # Ескі каналдарды өшіру for channel in target_guild.channels: try: await channel.delete() except: continue # Рөлдерді көшіру for role in reversed(source_guild.roles): if role.name != "@everyone": try: await target_guild.create_role(name=role.name, permissions=role.permissions, color=role.color) except: continue # Категориялар мен каналдар for category in source_guild.categories: new_cat = await target_guild.create_category(name=category.name) for channel in category.channels: if isinstance(channel, discord.TextChannel): await new_cat.create_text_channel(name=channel.name) elif isinstance(channel, discord.VoiceChannel): await new_cat.create_voice_channel(name=channel.name) await bot.close() except Exception as e: print(f"Қате шықты: {e}") await bot.close() @app.route('/') def index(): return render_template('index.html') @app.route('/clone', methods=['POST']) def clone(): data = request.json def run_it(): loop = asyncio.new_event_loop() asyncio.set_event_loop(loop) loop.run_until_complete(start_cloning(data['token'], data['source'], data['target'])) threading.Thread(target=run_it).start() return jsonify({"status": "Success", "message": "Процесс басталды! Аккаунтыңызды тексеріңіз."}) if __name__ == "__main__": app.run(host='0.0.0.0', port=8080)
