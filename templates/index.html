import discord from discord.ext import commands from flask import Flask, render_template, request, jsonify import threading app = Flask(__name__) async def start_cloning(token, source_id, target_id): intents = discord.Intents.all() bot = commands.Bot(command_prefix="!", intents=intents) @bot.event async def on_ready(): print(f'Бот қосылды: {bot.user}') source_guild = bot.get_guild(int(source_id)) target_guild = bot.get_guild(int(target_id)) if not source_guild or not target_guild: print("Сервер табылмады!") await bot.close() return print("Ескі каналдар мен рөлдер өшірілуде...") for channel in target_guild.channels: await channel.delete() for role in target_guild.roles: if role.name != "@everyone" and not role.managed: try: await role.delete() except: continue print("Рөлдер көшірілуде...") roles_map = {} for role in reversed(source_guild.roles): if role.name != "@everyone" and not role.managed: new_role = await target_guild.create_role(name=role.name, permissions=role.permissions, color=role.color, hoist=role.hoist, mentionable=role.mentionable) roles_map[role] = new_role print("Категориялар мен каналдар көшірілуде...") for category in source_guild.categories: new_category = await target_guild.create_category(name=category.name) for channel in category.channels: if isinstance(channel, discord.TextChannel): await new_category.create_text_channel(name=channel.name, topic=channel.topic, nsfw=channel.nsfw) elif isinstance(channel, discord.VoiceChannel): await new_category.create_voice_channel(name=channel.name, user_limit=channel.user_limit) print("Көшіру аяқталды!") await bot.close() try: await bot.start(token) except Exception as e: print(f"Қате: {e}") @app.route('/') def index(): return render_template('index.html') @app.route('/clone', methods=['POST']) def clone(): data = request.json # Ботты бөлек ағында іске қосу thread = threading.Thread(target=lambda: discord.utils.setup_logging() or discord.run_coroutine_threadsafe(start_cloning(data['token'], data['source'], data['target']), discord.utils._get_event_loop())) # Жеңілдетілген іске қосу нұсқасы def run_it(): import asyncio loop = asyncio.new_event_loop() asyncio.set_event_loop(loop) loop.run_until_complete(start_cloning(data['token'], data['source'], data['target'])) threading.Thread(target=run_it).start() return jsonify({"status": "Success", "message": "Процесс басталды! Серверіңізді тексеріңіз."}) if __name__ == "__main__": app.run(host='0.0.0.0', port=8080)
